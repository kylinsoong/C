= 回射服务

== 编译运行

[source, bash]
----
gcc server.c lib_utils.c lib_ios.c lib_socket.c str_echo.c -o echoserver
./echoserver

gcc client.c lib_utils.c lib_ios.c lib_socket.c str_cli.c -o echoclient
./echoclient
----
